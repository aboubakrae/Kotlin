fun main(args: Array<String>) {
    println("your name:")
    var name:String= readLine()!!.toString()
    println("your age:")
    var age:Int= readLine()!!.toInt()
    println("your tall:")
    var tall:Double= readLine()!!.toDouble()
    println("------Outputs------")
    println("name:"+name)
    println("age:"+age)
    println("tall:"+tall)
}
